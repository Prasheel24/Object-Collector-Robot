159 bool AStarPlanner::makePlan(const geometry_msgs::PoseStamped &start,
160  const geometry_msgs::PoseStamped &goal,
161  std::vector<geometry_msgs::PoseStamped> &plan) {
162  if (!initialized) {
163  ROS_ERROR_STREAM("The planner has not been initialized, please call"
164  << "initialize() to use the planner");
165  return false;
166  }
167  ROS_INFO_STREAM("Got a start " << start.pose.position.x << ", "
168  << start.pose.position.y << " and a goal:"
169  << goal.pose.position.x << ", "
170  << goal.pose.position.y);
171  plan.clear();
172  if (goal.header.frame_id != costmap_ros->getGlobalFrameID()) {
173  ROS_ERROR_STREAM("This planner as configured will only accept goals "
174  << " in the" << costmap_ros->getGlobalFrameID().c_str()
175  << " frame, but a goal was sent in the"
176  << goal.header.frame_id.c_str() << " frame.");
177  return false;
178  }
179  // take the start and goal to make plan
180  float startX = start.pose.position.x;
181  float startY = start.pose.position.y;
182 
183  float goalX = goal.pose.position.x;
184  float goalY = goal.pose.position.y;
185  convertToMapCoordinates(startX, startY); // sx - origin, sy -origin
186  convertToMapCoordinates(goalX, goalY); // gx - origin, gy - origin
187 
188  int startCell = 0;
189  int goalCell = 0;
190  // check if coordinates are within boundary
191  if (isCoordinateInBounds(startX, startY) &&
192  isCoordinateInBounds(goalX, goalY)) {
193  startCell = getCellIndex(startX, startY); // (x/res) * width + (y/res)
194  goalCell = getCellIndex(goalX, goalY);
195  } else {
196  ROS_WARN_STREAM("the start or goal is out of the map");
197  return false;
198  }
199 
200  if (isStartAndGoalValid(startCell, goalCell)) {
201  std::vector<int> bestPath;
202  bestPath.clear();
203  // main function call to run A star on the map
204  bestPath = runAStar(startCell, goalCell);
205  // run through map to make path
206  if (bestPath.size() > 0) {
207  for (auto& i : bestPath) {
208  float x = 0.0;
209  float y = 0.0;
210  float previous_x = 0.0;
211  float previous_y = 0.0;
212 
213  int index = i;
214  int previous_index;
215  getCellCoordinates(index, x, y);
216  if (i != 0) {
217  previous_index = (i-1);
218  } else {
219  previous_index = index;
220  }
221  getCellCoordinates(previous_index, previous_x, previous_y);
222  tf::Vector3 vectorToTarget;
223  vectorToTarget.setValue(x - previous_x, y - previous_y, 0.0);
224 
225  float angle = atan2(static_cast<double>(vectorToTarget.y()),
226  static_cast<double>(vectorToTarget.x()));
227  geometry_msgs::PoseStamped pose = goal;
228  pose.pose.position.x = x;
229  pose.pose.position.y = y;
230  pose.pose.position.z = 0.0;
231  pose.pose.orientation = tf::createQuaternionMsgFromYaw(angle);
232 
233  plan.push_back(pose);
234  }
235  return true;
236  } else {
237  ROS_WARN_STREAM("The planner failed to find a path" <<
238  " choose other goal position");
239  return false;
240  }
241  } else {
242  ROS_WARN_STREAM("Not valid start or goal");
243  return false;
244  }
245 }

297 std::vector<int> AStarPlanner::findPath(int startCell,
298  int goalCell,
299  std::vector<float> g_score) {
300  value++;
301  std::vector<int> bestPath;
302  std::vector<int> emptyPath;
303  GridSquare cell;
304  std::set<GridSquare> openSquaresList;
305  int currentCell;
306 
307  g_score[startCell] = 0;
308  cell.currentGridSquare = startCell;
309  cell.fCost = g_score[startCell] + calculateHCellScore(startCell, goalCell);
310  // open list to have free cells
311  openSquaresList.insert(cell);
312  currentCell = startCell;
313 
314  while (!openSquaresList.empty() && g_score[goalCell] == infinity) {
315  currentCell = openSquaresList.begin()->currentGridSquare;
316  openSquaresList.erase(openSquaresList.begin());
317  std::vector<int> neighborCells;
318  neighborCells = findFreeNeighborCell(currentCell);
319  for (auto& i : neighborCells) {
320  if (g_score[i] == infinity) {
321  g_score[i] = g_score[currentCell] +
322  getMoveToCellCost(currentCell,
323  i);
324  // add neighbor cells to list
325  addNeighborCellToOpenList(openSquaresList,
326  i,
327  goalCell, g_score);
328  }
329  }
330  }
331  if (g_score[goalCell] != infinity) {
332  // planner constructs path from here
333  bestPath = constructPath(startCell, goalCell, g_score);
334  return bestPath;
335  } else {
336  ROS_INFO_STREAM("Failure to find a path !");
337  return emptyPath;
338  }
339 }

342 std::vector<int> AStarPlanner::constructPath(int startCell,
343  int goalCell,
344  std::vector<float> g_score) {
345  std::vector<int> bestPath;
346  std::vector<int> path;
347 
348  path.insert(path.begin() + bestPath.size(), goalCell);
349  int currentCell = goalCell;
350  // construct a path that has minimum g and h function value
351  while (currentCell != startCell) {
352  std::vector<int> neighborCells;
353  neighborCells = findFreeNeighborCell(currentCell);
354  std::vector<float> gScoresNeighbors;
355  for (auto& i : neighborCells) {
356  gScoresNeighbors.push_back(g_score[i]);
357  }
358  std::vector<float>::iterator result =
359  std::min_element(gScoresNeighbors.begin(),
360  gScoresNeighbors.end());
361 
362  int posMinGScore = std::distance(gScoresNeighbors.begin(), result);
363  currentCell = neighborCells[posMinGScore];
364 
365  path.insert(path.begin() + path.size(), currentCell);
366  }
367  int p_size = path.size();
368  std::vector<int> tempPath = path;
369  auto it = 0;
370  // loop through to make path
371  for (auto& i : path) {
372  bestPath.insert(bestPath.begin() + bestPath.size(),
373  tempPath[(p_size - (it + 1))]);
374  ++it;
375  }
376  return bestPath;
377 }

392  int rowIndex = getCellRowIndex(cell);
393  int colIndex = getCellColIndex(cell);
394  int neighborIndex;
395  std::vector<int> freeNeighborCells = {0};
396  std::vector<int> row {-1, 0, 1};
397  std::vector<int> col {-1, 0, 1};
398  // check through the neighbor cells

400  for (auto& j : col) {

415  bool isvalid = true;

428  isvalid = false;
429  } else {

431  isvalid = false;
432  } else {
433  if (findFreeNeighborCell(startCell).size() == 0) {
434  isvalid = false;
435  }
436  }
437  }
438  }
439  }
440  }
441  return isvalid;
442 }

444 float AStarPlanner::getMoveToCellCost(int i1, int j1, int i2, int j2) {
445  float moveCost = infinity;
446  // if moving in diagonal, then h movecost is root 2
447  if ((j2 == j1 + 1 && i2 == i1 + 1) || (i2 == i1 - 1 && j2 == j1 + 1)
448  || (i2 == i1 - 1 && j2 == j1 - 1) || (j2 == j1 - 1 && i2 == i1 + 1)) {
449  moveCost = 1.4;
450  } else {
451  // if moving in straight line, then h movecost is 1
452  if ((j2 == j1 && i2 == i1 - 1) || (i2 == i1 && j2 == j1 - 1)
453  || (i2 == i1 + 1 && j2 == j1) || (i2 == i1 && j2 == j1 + 1)) {
454  moveCost = 1;
455  }
456  }
457  return moveCost;
458 }
